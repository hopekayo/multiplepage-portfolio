<h4>Agent 自主性监控</h4>
<pre><code class="language-python">class AgentAutonomyMonitor:
    def __init__(self):
        self.metrics = {
            &quot;tasks_completed&quot;: 0,
            &quot;human_interventions&quot;: 0,
            &quot;errors_recovered&quot;: 0,
            &quot;total_decisions&quot;: 0
        }
    
    def log_task(self, success: bool, intervention: bool = False):
        self.metrics[&quot;tasks_completed&quot;] += 1
        if intervention:
            self.metrics[&quot;human_interventions&quot;] += 1
    
    def log_error_recovery(self, recovered: bool):
        if recovered:
            self.metrics[&quot;errors_recovered&quot;] += 1
    
    def calculate_autonomy_score(self) -&gt; float:
        if self.metrics[&quot;tasks_completed&quot;] == 0:
            return 0.0
        
        completion_rate = 1.0  # 假设所有任务都完成
        intervention_rate = self.metrics[&quot;human_interventions&quot;] / self.metrics[&quot;tasks_completed&quot;]
        recovery_rate = self.metrics[&quot;errors_recovered&quot;] / max(1, self.metrics[&quot;tasks_completed&quot;] - self.metrics[&quot;errors_recovered&quot;])
        
        return (
            completion_rate * 0.3 +
            (1 - intervention_rate) * 0.25 +
            recovery_rate * 0.2 +
            0.15 +  # 假设决策置信度
            0.1     # 假设上下文保持
        )
</code></pre>
