<p><strong>考察点：</strong></p>
<ul>
<li>Promise.all 任一失败则整体失败</li>
<li>Promise.allSettled 等待所有完成，返回状态数组</li>
<li>手写实现考察对 Promise 的理解</li>

<pre><code class="language-javascript">function promiseAll(promises) {
  return new Promise((resolve, reject) =&gt; {
    const results = [];
    let completed = 0;
    
    if (promises.length === 0) {
      resolve(results);
      return;
    }
    
    promises.forEach((promise, index) =&gt; {
      Promise.resolve(promise).then(result =&gt; {
        results[index] = result;
        completed++;
        if (completed === promises.length) resolve(results);
      }).catch(reject);
    });
  });
}
</code></pre>

  if (obj === null || typeof obj !== &#39;object&#39;) return obj;
  if (obj instanceof Date) return new Date(obj);
  if (obj instanceof RegExp) return new RegExp(obj);
  if (hash.has(obj)) return hash.get(obj);
  
  const clone = Array.isArray(obj) ? [] : {};
  hash.set(obj, clone);
  
  for (let key in obj) {
    if (obj.hasOwnProperty(key)) {
      clone[key] = deepClone(obj[key], hash);
    }
  }
  return clone;

function debounce(fn, delay) {
  let timer = null;
  return function(...args) {
    clearTimeout(timer);
    timer = setTimeout(() =&gt; fn.apply(this, args), delay);
  };
}

// 节流
function throttle(fn, delay) {
  let lastTime = 0;
  return function(...args) {
    const now = Date.now();
    if (now - lastTime &gt;= delay) {
      lastTime = now;
      fn.apply(this, args);
    }
  };
}
</code></pre>

async function requestWithLimit(tasks, limit) {
  const results = [];
  const running = [];
  
  for (let i = 0; i &lt; tasks.length; i++) {
    const p = tasks[i]().then(res =&gt; {

class AsyncQueue {
  constructor(limit) {
    this.limit = limit;
    this.queue = [];
    this.running = 0;
  }
  
  add(task) {
    return new Promise((resolve, reject) =&gt; {
      this.queue.push({ task, resolve, reject });
      this.run();
    });
  }
  
  run() {
    while (this.running &lt; this.limit &amp;&amp; this.queue.length) {
      const { task, resolve, reject } = this.queue.shift();
      this.running++;
      task().then(resolve).catch(reject).finally(() =&gt; {
        this.running--;
        this.run();
